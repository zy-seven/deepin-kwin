    main.cpp
    kwincompositingdata.cpp
)

)

    OUTPUT_NAME deepin-kwincompositing
)

    Qt::DBus
    Qt::Widgets

    KF6::ConfigCore
    KF6::CoreAddons
    KF6::I18n
    KF6::KCMUtils
    KF6::WindowSystem
